def repair_coupang_orders_parquet() -> str:
    """기존 orders parquet의 N중복 행을 (store_id, order_id, menu_options) 기준으로 제거."""
    files = sorted(COUPANG_ORDERS_DB.rglob("orders_*.parquet"))
    if not files:
        return f"수정 대상 없음 | {COUPANG_ORDERS_DB}"

    total_before = 0
    total_after = 0
    for path in files:
        try:
            df = pd.read_parquet(path)
        except Exception as exc:
            logger.warning("parquet 로드 실패, 스킵: %s | %s", path, exc)
            continue
        before = len(df)
        total_before += before
        dedup_cols = [c for c in ["store_id", "order_id", "menu_options"] if c in df.columns]
        if dedup_cols:
            df = (
                df.sort_values("collected_at", ascending=False, na_position="last")
                .drop_duplicates(subset=dedup_cols, keep="first")
                .reset_index(drop=True)
            )
        after = len(df)
        total_after += after
        if after < before:
            df.to_parquet(path, index=False)
            logger.info("repair: %s | %d → %d행 (%d 제거)", path.name, before, after, before - after)
        else:
            logger.info("변경 없음: %s (%d행)", path.name, after)

    removed = total_before - total_after
    return f"수정 완료 | 파일={len(files)} 제거={removed}행 ({total_before}→{total_after})"
```

def reconcile_coupang_for_test_stores(
    stores: list[str],
    sale_date: str | None = None,
    lookback_days: int = 7,
) -> str:
    """TEST_STORES의 쿠팡이츠 행을 coupang_macro 직수집 기준으로 교정."""
    if sale_date:
        dates = [str(sale_date)]
    else:
        kst_now = pendulum.now("Asia/Seoul")
        dates = [
            (kst_now - timedelta(days=i)).strftime("%Y-%m-%d")
            for i in range(lookback_days)
        ]
    ym_list = sorted({d[:7] for d in dates})

    total_added = 0
    total_removed = 0
    store_map = _load_store_map()

    for store in stores:
        for ym in ym_list:
            coupang_files = _find_coupang_files(store, ym)
            opt_map = _find_options_map(store, ym)

            if not coupang_files:
                for date in dates:
                    if date[:7] != ym:
                        continue
                    removed, added = _upsert_daily(pd.DataFrame(columns=UNIFIED_COLUMNS), date, store)
                    total_removed += removed
                    total_added += added
                    logger.info(
                        "쿠팡 수집 파일 없음, 기존 쿠팡이츠 행 제거: store=%s date=%s | 제거=%d 추가=%d",
                        store, date, removed, added,
                    )
                continue

            df_raw = _read_coupang_files(coupang_files)
            if df_raw.empty:
                for date in dates:
                    if date[:7] != ym:
                        continue
                    removed, added = _upsert_daily(pd.DataFrame(columns=UNIFIED_COLUMNS), date, store)
                    total_removed += removed
                    total_added += added
                continue

            # dedup: 수집 버그 방어 (다른 컴퓨터 중복 수집)
            dedup_cols = [c for c in ["order_id", "menu_options"] if c in df_raw.columns]
            if dedup_cols:
                df_raw = (
                    df_raw.sort_values("collected_at", ascending=False, na_position="last")
                    .drop_duplicates(subset=dedup_cols, keep="first")
                    .reset_index(drop=True)
                )

            brand = _parse_brand_from_path(coupang_files[0])
            parsed = df_raw["order_date"].map(_parse_coupang_datetime)
            df_raw = df_raw.copy()
            df_raw["sale_date"] = parsed.map(lambda v: v[0])
            df_raw["order_time_str"] = parsed.map(lambda v: v[1])

            for date in dates:
                if date[:7] != ym:
                    continue
                df_day = df_raw[df_raw["sale_date"] == date].copy()
                if df_day.empty:
                    removed, added = _upsert_daily(pd.DataFrame(columns=UNIFIED_COLUMNS), date, store)
                    total_removed += removed
                    total_added += added
                    continue

                df_unified = _transform_to_unified(df_day, store, brand, store_map, opt_map)
                if df_unified.empty:
                    removed, added = _upsert_daily(df_unified, date, store)
                    total_removed += removed
                    total_added += added
                    continue
                df_unified = _recalculate_order_fields(df_unified)

                removed, added = _upsert_daily(df_unified, date, store)
                total_removed += removed
                total_added += added
                logger.info(
                    "쿠팡 교정 완료: store=%s date=%s | 제거=%d 추가=%d",
                    store, date, removed, added,
                )

    return f"쿠팡수동 교정 완료 | 제거={total_removed}행 추가={total_added}행"


def enforce_coupang_manual_only_for_test_stores(
    stores: list[str],
    sale_date: str | None = None,
    lookback_days: int = 7,
) -> str:
    """TEST_STORES의 쿠팡이츠 최종 방어막.

    platform=쿠팡이츠 행 중 source!=쿠팡수동인 행을 제거한다.
    """
    if not stores:
        return "TEST_STORES 없음 - 쿠팡수동 최종 정리 스킵"

    if sale_date:
        dates = [str(sale_date)]
    else:
        kst_now = pendulum.now("Asia/Seoul")
        dates = [
            (kst_now - timedelta(days=i)).strftime("%Y-%m-%d")
            for i in range(lookback_days)
        ]

    store_set = {str(s).strip() for s in stores if str(s).strip()}
    total_removed = 0
    changed_files = 0

    for date in dates:
        daily_path = _unified_daily_path(date)
        if not daily_path.exists():
            continue

        df = pd.read_parquet(daily_path).reindex(columns=UNIFIED_COLUMNS, fill_value="")
        remove_mask = (
            df["store"].fillna("").astype(str).str.strip().isin(store_set)
            & df["platform"].fillna("").astype(str).str.strip().eq(COUPANG_PLATFORM)
            & ~df["source"].fillna("").astype(str).str.strip().eq(COUPANG_SOURCE)
        )
        removed = int(remove_mask.sum())
        if removed == 0:
            continue

        df_out = df[~remove_mask].reset_index(drop=True)
        for col in ("qty", "unit_price", "total_price", "discount_amount", "order_cnt"):
            if col in df_out.columns:
                df_out[col] = pd.to_numeric(df_out[col], errors="coerce").fillna(0).astype(int)
        df_out = df_out.reindex(columns=UNIFIED_COLUMNS, fill_value="")
        df_out.to_parquet(daily_path, index=False, engine="pyarrow")

        total_removed += removed
        changed_files += 1
        logger.warning(
            "쿠팡수동 최종 정리: %s | 제거=%d | stores=%s",
            daily_path.name, removed,
            sorted(df.loc[remove_mask, "store"].dropna().astype(str).unique().tolist()),
        )

    return f"쿠팡수동 최종 정리 완료 | 파일={changed_files} 제거={total_removed}행"

def _find_coupang_files(store: str, ym: str) -> list[str]:
    base = ANALYTICS_DB / "coupang_macro" / "orders"
    pattern = str(base / "brand=*" / f"store={store}" / f"ym={ym}" / f"orders_{ym}.parquet")
    return sorted(glob(pattern))


def _find_options_map(store: str, ym: str) -> dict[str, int]:
    """options.csv → {옵션명: option_price}. 파일 없으면 {}."""
    base = ANALYTICS_DB / "coupang_macro" / "options"
    pattern = str(base / "brand=*" / f"store={store}" / f"ym={ym}" / "options.csv")
    files = sorted(glob(pattern))
    if not files:
        return {}
    try:
        df = pd.read_csv(files[0], dtype=str, encoding="utf-8-sig")
        df["option_price"] = pd.to_numeric(df["옵션가격"], errors="coerce").fillna(0).astype(int)
        return {str(r["옵션명"]).strip(): r["option_price"] for _, r in df.iterrows()}
    except Exception as exc:
        logger.warning("options.csv 로드 실패: %s | %s", files[0], exc)
        return {}


def _read_coupang_files(files: list[str]) -> pd.DataFrame:
    dfs = []
    for path in files:
        try:
            df = pd.read_parquet(path)
            df["_src_path"] = path
            dfs.append(df)
        except Exception as exc:
            logger.warning("쿠팡 파일 읽기 실패: %s | %s", path, exc)
    return pd.concat(dfs, ignore_index=True) if dfs else pd.DataFrame()


def _parse_brand_from_path(file_path: str) -> str:
    match = re.search(r"brand=([^/\\]+)", file_path)
    return match.group(1) if match else ""


def _parse_coupang_datetime(value) -> tuple[str, str]:
    """'2026.06.15 22:35' → ('2026-06-15', '22:35:00')."""
    if not isinstance(value, str):
        return "", ""
    match = re.match(r"(\d{4})\.(\d{2})\.(\d{2})\s+(\d{1,2}):(\d{2})", str(value).strip())
    if not match:
        return "", ""
    year, month, day = match.group(1), match.group(2), match.group(3)
    hour, minute = int(match.group(4)), int(match.group(5))
    return f"{year}-{month}-{day}", f"{hour:02d}:{minute:02d}:00"


def _calc_total_price(order_df: pd.DataFrame, sales: float, opt_map: dict[str, int]) -> pd.Series:
    """
    2단계 배분:
    1단계: menu_name 그룹별로 menu_price * qty 비율로 sales 배분
    2단계: 그룹 내 opt_price > 0 행은 opt_price 비율, main row는 나머지 흡수, 미매칭/0원 행은 0
    """
    if pd.isna(sales) or sales == 0:
        result = pd.Series(0, index=order_df.index)
        if not order_df.empty:
            result.iloc[0] = 0
        return result

    result = pd.Series(0, index=order_df.index, dtype=int)
    groups = list(order_df.groupby("menu_name", sort=False))

    # 1단계: 그룹 weight = menu_price * qty (항상 menu_price 기준)
    group_weights: list[tuple] = []
    for menu_name, grp in groups:
        mp = pd.to_numeric(grp["menu_price"].iloc[0] if not grp.empty else 0, errors="coerce")
        qty = pd.to_numeric(grp["menu_qty"].iloc[0] if not grp.empty else 1, errors="coerce") or 1
        weight = float(mp * qty) if pd.notna(mp) else 0.0
        group_weights.append((menu_name, grp, weight))

    total_w = sum(w for _, _, w in group_weights)
    if total_w == 0:
        result.iloc[0] = int(sales)
        return result

    # 1단계 배분
    remaining = int(sales)
    for i, (menu_name, grp, weight) in enumerate(group_weights):
        if i < len(group_weights) - 1:
            group_sales = round(weight / total_w * sales)
        else:
            group_sales = remaining
        remaining -= group_sales

        # 2단계: opt_price 기반 인트라-그룹 배분
        opt_prices = grp["menu_options"].map(
            lambda o: opt_map.get(str(o).strip(), 0) if pd.notna(o) else 0
        )
        matched_mask = opt_prices > 0
        # main row = menu_price가 notna인 첫 행
        mp_vals = pd.to_numeric(grp["menu_price"], errors="coerce")
        main_idx = mp_vals[mp_vals.notna()].index[0] if mp_vals.notna().any() else grp.index[0]

        row_alloc = pd.Series(0, index=grp.index, dtype=int)
        if matched_mask.any():
            opt_sum = int(opt_prices[matched_mask].sum())
            for idx in grp[matched_mask].index:
                row_alloc.loc[idx] = round(opt_prices.loc[idx] / opt_sum * group_sales)
            allocated_to_opts = int(row_alloc.sum())
            row_alloc.loc[main_idx] = group_sales - allocated_to_opts
        else:
            row_alloc.loc[main_idx] = group_sales

        for idx, val in row_alloc.items():
            result.loc[idx] = int(val)

    return result


def _transform_to_unified(
    df: pd.DataFrame,
    store: str,
    brand: str,
    store_map: dict,
    opt_map: dict[str, int],
) -> pd.DataFrame:
    """coupang_macro rows → UNIFIED_COLUMNS DataFrame."""
    # 취소 주문 제외 (is_cancelled == "Y" 이거나 매출액 <= 0)
    is_cancelled = df["is_cancelled"].fillna("N").astype(str).str.strip().str.upper() if "is_cancelled" in df.columns else pd.Series("N", index=df.index)
    sales_raw = pd.to_numeric(df["매출액"], errors="coerce").fillna(0) if "매출액" in df.columns else pd.Series(0, index=df.index)
    active_mask = (is_cancelled != "Y") & (sales_raw > 0)
    df = df[active_mask].copy()
    if df.empty:
        return pd.DataFrame(columns=UNIFIED_COLUMNS)

    out = pd.DataFrame(index=df.index)

    out["sale_date"] = df["sale_date"]
    out["ym"] = df["sale_date"].str[:7]
    out["source"] = COUPANG_SOURCE
    out["brand"] = brand
    out["store"] = store
    out["platform"] = COUPANG_PLATFORM
    out["order_id"] = df["order_id"].fillna("").astype(str).str.strip()
    out["order_time"] = df["order_time_str"]
    out["sale_type"] = "정상"
    out["order_type"] = df["delivery_type"].map(
        lambda v: "배달_포장" if str(v).strip() == "포장" else "배달"
    ) if "delivery_type" in df.columns else "배달"
    out["menu_name"] = (
        df["order_summary"].fillna("").astype(str)
        .str.replace(r"\s*외\s*\d+건$", "", regex=True)
        .str.strip()
    ) if "order_summary" in df.columns else df["menu_name"].fillna("").astype(str)
    out["item_name"] = df["menu_options"].fillna("").astype(str).str.strip() if "menu_options" in df.columns else ""
    out["qty"] = _to_int_series(df["menu_qty"]).replace(0, 1) if "menu_qty" in df.columns else 1
    out["discount_amount"] = 0

    # total_price: 주문별 2단계 배분
    total_price_col = pd.Series(0, index=df.index, dtype=int)
    for oid in out["order_id"].unique():
        oid_mask = out["order_id"] == oid
        sales_val = float(sales_raw.loc[oid_mask].iloc[0]) if oid_mask.any() else 0
        sub = df.loc[oid_mask].copy()
        sub["menu_name"] = out.loc[oid_mask, "menu_name"]
        allocated = _calc_total_price(sub, sales_val, opt_map)
        total_price_col.loc[oid_mask] = allocated.values

    out["total_price"] = total_price_col
    out["unit_price"] = out["total_price"]  # item-level; 하나씩이므로 동일

    out["item_seq"] = out.groupby("order_id").cumcount().add(1).astype(int).astype(str)
    out["item_id"] = out["item_name"].map(
        lambda v: hashlib.md5(v.encode()).hexdigest() if v else ""
    )
    out["담당자"] = _lookup_store_meta(store_map, store, "담당자")
    out["region"] = _lookup_store_meta(store_map, store, "region")
    out["실오픈일"] = _lookup_store_meta(store_map, store, "실오픈일")
    out["collected_at"] = pendulum.now("Asia/Seoul").isoformat()

    out = _recalculate_order_fields(out)
    return out.reindex(columns=UNIFIED_COLUMNS, fill_value="")


def _recalculate_order_fields(df: pd.DataFrame) -> pd.DataFrame:
    """order_cnt / _pk 재계산."""
    if df.empty:
        return df
    df = df.copy()
    df["order_cnt"] = 0

    representative_indexes = []
    for _, group in df.groupby("order_id", sort=False):
        menu_match = group["item_name"].fillna("").astype(str).str.strip().eq(
            group["menu_name"].fillna("").astype(str).str.strip()
        )
        if menu_match.any():
            representative_indexes.append(group[menu_match].index[0])
            continue
        unit_price = pd.to_numeric(group["unit_price"], errors="coerce").fillna(0)
        representative_indexes.append(unit_price.idxmax())

    if representative_indexes:
        df.loc[representative_indexes, "order_cnt"] = df.loc[representative_indexes, "sale_type"].map(
            lambda v: 0 if str(v) == "취소" else 1
        )

    df["_pk"] = _make_unified_pk(df)
    return df


def _upsert_daily(df_new: pd.DataFrame, date: str, store: str) -> tuple[int, int]:
    """unified_sales 일별 parquet에서 해당 store의 쿠팡이츠 행을 교체."""
    UNIFIED_ROOT.mkdir(parents=True, exist_ok=True)
    daily_path = _unified_daily_path(date)

    if daily_path.exists():
        df_existing = pd.read_parquet(daily_path).reindex(columns=UNIFIED_COLUMNS, fill_value="")
        remove_mask = (
            df_existing["store"].fillna("").astype(str).str.strip().eq(store)
            & df_existing["platform"].fillna("").astype(str).str.strip().eq(COUPANG_PLATFORM)
        )
        removed_count = int(remove_mask.sum())
        df_existing = df_existing[~remove_mask]
    else:
        df_existing = pd.DataFrame(columns=UNIFIED_COLUMNS)
        removed_count = 0

    df_out = pd.concat([df_existing, df_new], ignore_index=True)
    if "_pk" in df_out.columns:
        before = len(df_out)
        df_out = df_out.drop_duplicates(subset=["_pk"], keep="last").reset_index(drop=True)
        dropped = before - len(df_out)
        if dropped:
            logger.warning("쿠팡 교정 저장 중 _pk 중복 %d행 제거", dropped)

    for col in ("qty", "unit_price", "total_price", "discount_amount", "order_cnt"):
        if col in df_out.columns:
            df_out[col] = pd.to_numeric(df_out[col], errors="coerce").fillna(0).astype(int)

    df_out = df_out.reindex(columns=UNIFIED_COLUMNS, fill_value="")
    df_out.to_parquet(daily_path, index=False, engine="pyarrow")
    return removed_count, len(df_new)
```

## Constraints

- `_recalculate_order_fields` 는 `DB_UnifiedSales_baemin.py`에 동일 코드가 있으므로 **복사** 사용 (common.py 이동 없이)
- `_upsert_daily` 내부에서 `platform==COUPANG_PLATFORM` 으로 교체 (배민은 BAEMIN_PLATFORM 사용 → 독립적)
- `collected_at` 컬럼이 orders parquet에 없을 수 있음 → `sort_values("collected_at", ...)` 시 KeyError 방어 필요 (`if "collected_at" in df.columns` 확인 후 sort)
- `LOOKBACK_DAYS` 가 `None` 인 경우 `LOOKBACK_DAYS or 7` 로 처리 (기존 baemin 패턴 동일)
- `is_cancelled` 컬럼 없을 수 있음 → `if "is_cancelled" in df.columns` 방어 처리
- 기존 `_row_hash` dedup은 완전히 제거하지 말고, dedup_cols 없을 때만 fallback으로 사용
