### runner.js — 재시도 라운드가 blocked 매장을 제외하는 지점 (동작 근거)
```js
function markAccountStoresBlocked(acc, targetStores, note) {
  targetStores.forEach(store => {
    const row = ensureStoreStatus(store, acc);
    ['orders', 'cmg', 'menu'].forEach(category => {
      row[`${category}Tried`] = true;
      row[`${category}Ok`] = true;
    });
    row.blocked = true;               // ← 이 플래그 때문에 재시도에서 제외됨
    row.note = note || row.note || '';
    row.collected_at = new Date().toISOString();
  });
}
// incompleteStores(): 재시도 대상 = blocked 아니고 아직 미완료인 매장
function incompleteStores() {
  return [...storeStatusRows.values()]
    .filter(r => !r.blocked && !(r.ordersOk && r.cmgOk && r.menuOk))  // ← !r.blocked
    .map(r => r.store);
}
// markAccountStoresTried(): blocked=false, Ok=false 유지 → incompleteStores에 포함(재시도됨)
function markAccountStoresTried(acc, targetStores, note) {
  targetStores.forEach(store => {
    const row = ensureStoreStatus(store, acc);
    ['orders', 'cmg', 'menu'].forEach(category => { row[`${category}Tried`] = true; });
    row.note = note || row.note || '';
    row.collected_at = new Date().toISOString();
  });
}
```
