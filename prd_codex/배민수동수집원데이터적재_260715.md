```python
import re

_KNOWN_BRANDS = ("나홀로", "도리당")

def _baemin_now_brand_store(store_name: str) -> tuple[str, str]:
    """Selenium NOW(_parse_store_option)와 동일 규칙: (brand, store)."""
    text = re.sub(r"\[.*?\]\s*", "", str(store_name or "")).strip()
    brand = next((b for b in _KNOWN_BRANDS if b in text), "")
    matches = re.findall(r"[가-힣A-Za-z0-9]+(?:점|지점|분점|직영점)", text)
    store = matches[-1] if matches else (text.split()[-1] if text.split() else "")
    return brand, store

def _date_from_metrics(path: Path, collected_at: str) -> str:
    """파일명 마지막 8자리 우선(store_id도 8자리라 반드시 [-1]), 없으면 collected_at 날짜."""
    tokens = re.findall(r"\d{8}", path.name)
    if tokens:
        t = tokens[-1]
        return f"{t[:4]}-{t[4:6]}-{t[6:]}"
    return str(collected_at)[:10]
```
**주의:** 주문 로더의 `_manual_baemin_store_meta`(정규화·strip_brand)는 marketing/metrics에 쓰지 말 것 — 폴더 키가 어긋난다.

```python
def _load_csv_dataset(files, out_dir, out_name, ym_fn, dedup_subset) -> dict:
    grouped_rows: dict[tuple[str,str,str], list[dict]] = {}
    for item in files:
        path = Path(item["path"]); source = item.get("source","unknown")
        df = pd.read_csv(path, encoding="utf-8-sig", dtype=str)
        if df.empty: continue
        for _, row in df.iterrows():
            brand, store = _resolve_brand_store(row.get("매장명",""))
            if not brand or not store: continue
            ym = ym_fn(row, path)
            if not ym: continue
            grouped_rows.setdefault((brand,store,ym), []).append(row.astype(str).to_dict())
    for (brand,store,ym), rows in grouped_rows.items():
        target_dir = out_dir / f"brand={brand}" / f"store={store}" / f"ym={ym}"
        target_dir.mkdir(parents=True, exist_ok=True)
        out_path = target_dir / f"{out_name}.csv"
        new_df = pd.DataFrame(rows)
        combined = pd.concat([pd.read_csv(out_path, dtype=str), new_df], ignore_index=True) if out_path.exists() else new_df
        combined = combined.astype(str)
        for col in [c for c in dedup_subset if c not in combined.columns]: combined[col] = ""
        if dedup_subset: combined = combined.drop_duplicates(subset=dedup_subset, keep="last")
        combined.to_csv(out_path, index=False, encoding="utf-8-sig")

def _ym_from_filename(path: Path) -> str:
    matches = re.findall(r"\d{8}", path.name)
    token = matches[-1]                     # store_id도 8자리 → 반드시 마지막
    return f"{token[:4]}-{token[4:6]}"
```

### db_baemin_marketing.py (재사용할 파서)
```python
def _extract_branch(store_name: str) -> str:
    # '[음식배달] ... 도리당 김포장기점' → '김포장기점'
    name = re.sub(r"\[.*?\]\s*", "", str(store_name)).strip()
    tokens = name.split()
    return tokens[-1] if tokens else name

def _extract_brand_from_filename(filename: str) -> str:
    # baemin_marketing_..._도리당_초월점_14552300_202603.csv → '도리당' (곱도리탕→나홀로)
    stem = Path(filename).stem; parts = stem.split('_')
    for i, part in enumerate(parts):
        if re.match(r'^\d{6,}$', part) and i >= 2:
            brand = parts[i-2]
            return "나홀로" if brand == "곱도리탕" else brand
    return "unknown"
```

## Constraints
- **파티션 키는 기존 폴더와 반드시 일치.** marketing=`_extract_branch`+`_extract_brand_from_filename`, metrics=`_baemin_now_brand_store`. 주문용 `_manual_baemin_store_meta` 금지.
- `db_baemin_marketing.save_partitioned_csv` 의 `_load_legacy_data`(shutil.rmtree 마이그레이션)는 **재사용 금지** — 단순 idempotent 저장만.
- 성공(1행 이상 적재) 파일만 아카이브, 파싱 실패 파일은 폴더에 잔존.
- 인코딩 utf-8-sig, 전 컬럼 문자열(dtype=str).
- 쿠팡 로더(`DB_CoupangMacro_load`)·주간 marketing DAG·Selenium NOW 경로는 **수정 금지**.
- 태스크 id·DAG 의존성(`>>`) 변경 금지.
