```javascript
_getEarliestOrderDate(rows) {
  if (!rows || rows.length === 0) return null;
  return rows.reduce((min, r) => {
    const d = r.order_date || '';
    return (d && d < min) ? d : min;
  }, rows[0].order_date || '');
},
```

### 3. 날짜 범위 UI 재설정 헬퍼 `_setOrdersDateRange` 추가

`_setOrdersDateToYesterday` 함수 바로 다음에 추가한다.
`_setOrdersDateToYesterday`의 DayPicker 클릭 패턴을 그대로 쓰되, startDate와 endDate를 별도 Date 객체로 받는다.

```javascript
// startDateObj, endDateObj: Date 객체
async _setOrdersDateRange(startDateObj, endDateObj) {
  const makeLabel = (d) =>
    `${d.getFullYear()}.${d.getMonth() + 1}.${d.getDate()}`;
  const makeTimestamp = (d) => String(new Date(d.getFullYear(), d.getMonth(), d.getDate()).getTime());

  const startLabel = makeLabel(startDateObj);
  const endLabel   = makeLabel(endDateObj);
  const startTs    = makeTimestamp(startDateObj);
  const endTs      = makeTimestamp(endDateObj);
  const startAriaLabel = startDateObj.toDateString();
  const endAriaLabel   = endDateObj.toDateString();

  const getVisibleDateInputs = () =>
    [...document.querySelectorAll('div[data-testid="input"]')]
      .filter(el => el.offsetParent !== null && /\d{4}\.\d{1,2}\.\d{1,2}/.test(el.textContent || ''));

  const nativeSetter = Object.getOwnPropertyDescriptor(window.HTMLInputElement.prototype, 'value')?.set;
  const forceSetHidden = () => {
    const pairs = [['startDate', startTs], ['endDate', endTs]];
    for (const [name, val] of pairs) {
      const el = document.querySelector(`input[name="${name}"]`);
      if (!el) continue;
      if (nativeSetter) nativeSetter.call(el, val); else el.value = val;
      el.setAttribute('value', val);
      el.dispatchEvent(new Event('input', { bubbles: true }));
      el.dispatchEvent(new Event('change', { bubbles: true }));
    }
    const inputs = getVisibleDateInputs();
    if (inputs[0]) { inputs[0].textContent = startLabel; inputs[0].dispatchEvent(new Event('input', { bubbles: true })); }
    if (inputs[1]) { inputs[1].textContent = endLabel;   inputs[1].dispatchEvent(new Event('input', { bubbles: true })); }
  };

  const waitForDayPicker = async (ms = 3000) => {
    const t = Date.now();
    while (Date.now() - t < ms) {
      if (document.querySelector('div.DayPicker-Day')) return true;
      await new Promise(r => setTimeout(r, 100));
    }
    return false;
  };

  const clickDayCell = async (ariaLabel, dayObj) => {
    const dayNum = String(dayObj.getDate());
    let cells = [...document.querySelectorAll(`div.DayPicker-Day[aria-label="${ariaLabel}"]`)]
      .filter(c => c.getAttribute('aria-disabled') !== 'true' && !c.classList.contains('DayPicker-Day--outside'));
    if (!cells.length)
      cells = [...document.querySelectorAll('div.DayPicker-Day')]
        .filter(c => c.textContent.trim() === dayNum && !c.classList.contains('DayPicker-Day--outside') && c.getAttribute('aria-disabled') !== 'true');
    if (!cells.length) return false;
    cells[cells.length - 1].click();
    return true;
  };

  // 날짜 picker 열기
  const trigger = document.querySelector('div.e4pgcj08') ||
    [...document.querySelectorAll('div')].find(el =>
      el.offsetParent !== null && el.querySelectorAll('div[data-testid="input"]').length >= 2);
  if (trigger) { trigger.click(); await this._randomDelay(400, 600); }

  let dateInputs = getVisibleDateInputs();
  const startInput = dateInputs[0] || document.querySelector('div[data-testid="input"]');
  if (!startInput) { forceSetHidden(); await this._randomDelay(300, 400); return true; }

  startInput.click(); await this._randomDelay(300, 400);
  if (!await waitForDayPicker()) { forceSetHidden(); return true; }

  // startDate 달 이동 (현재 달보다 이전이면 prev 버튼)
  // (단순 버전: aria-label 매칭 실패 시 prev 한 번 시도)
  if (!await clickDayCell(startAriaLabel, startDateObj)) {
    const prevBtn = document.querySelector('.DayPicker-NavButton--prev');
    if (prevBtn) { prevBtn.click(); await this._randomDelay(300, 400); }
    await clickDayCell(startAriaLabel, startDateObj);
  }
  await this._randomDelay(400, 500);

  dateInputs = getVisibleDateInputs();
  const endInput = dateInputs[1] || document.querySelector('.e1mdtx7j2 [data-testid="input"]');
  if (endInput) {
    endInput.click(); await this._randomDelay(300, 400);
    if (await waitForDayPicker()) {
      if (!await clickDayCell(endAriaLabel, endDateObj)) {
        const nextBtn = document.querySelector('.DayPicker-NavButton--next');
        if (nextBtn) { nextBtn.click(); await this._randomDelay(300, 400); }
        await clickDayCell(endAriaLabel, endDateObj);
      }
    }
  }

  forceSetHidden(); // 보험: hidden input도 강제 설정
  await this._randomDelay(400, 600);
  return true;
},
```

### 4. 재시작 핵심 함수 `_restartFromErrorDate` 추가

```javascript
// 에러 발생 시 에러 날짜~원래 endDate로 날짜 범위 재설정 후 1페이지부터 재수집
// safeRows: 에러 직전까지 수집된 전체 rows
// shopInfo: 매장 정보
// originalEndDateMs: 원래 endDate (epoch ms 문자열, _readOrdersDateRange().end)
async _restartFromErrorDate(safeRows, shopInfo, originalEndDateMs) {
  if (this._collectSource !== 'manual') return { restarted: false };

  const errorDateStr = this._getEarliestOrderDate(safeRows);
  if (!errorDateStr) return { restarted: false };

  // 에러 날짜 이전 rows만 보존 (에러 날짜는 재수집 대상)
  const preRows = safeRows.filter(r => (r.order_date || '') < errorDateStr);

  const endDateMs = parseInt(originalEndDateMs, 10);
  const endDateObj = isNaN(endDateMs) ? new Date() : new Date(endDateMs);
  const startDateObj = new Date(errorDateStr); // "YYYY-MM-DD" 파싱

  Utils.updateProgressModal({
    debug: `⚠️ 에러 발생 → ${errorDateStr} ~ ${endDateObj.toLocaleDateString('ko-KR')} 재조회 시작`
  });

  const ok = await this._setOrdersDateRange(startDateObj, endDateObj);
  if (!ok) return { restarted: false };

  await this._clickOrdersSearchButton();
  await this._waitForOrdersDataReady(10000);

  const seed = {
    allRows: preRows,
    collectedOrderIds: new Set(preRows.map(r => r.order_id).filter(Boolean)),
    pageCount: 0,
    _originalEndDateMs: originalEndDateMs,
  };
  const { allRows: newRows, hasError } = await this._collectOrdersPages(shopInfo, seed);
  return { restarted: true, newRows, hasError };
},
```

3. **rows 병합 확인**: 재시작 후 최종 CSV에 에러 전 rows(06-01~06-14) + 재수집 rows(06-15~06-30) 모두 포함 확인

4. **중복 없음 확인**: `collectedOrderIds` Set에 의해 동일 order_id가 2번 들어가지 않는지 확인

## Constraints

- `_collectSource !== 'manual'` 인 경우(배치) `_restartFromErrorDate`가 즉시 `{ restarted: false }` 반환해야 함 — 배치 동작 변경 금지
- `_hardReloadResume` 함수 자체는 삭제하지 말 것 — `_resumeOrdersFromReload`에서 여전히 참조 가능
- `order_date` 필드가 없는 row는 재시작 기준에서 제외 (filter로 무시)
- `_setOrdersDateRange`에서 달력 월 이동은 최대 1회만 시도 — 무한 루프 방지
- 재시작 중 `_stopFlag`가 true이면 즉시 중단
