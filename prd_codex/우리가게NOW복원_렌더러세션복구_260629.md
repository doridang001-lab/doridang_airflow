def _is_session_issue(exc: Exception) -> bool:
    error_text = str(exc).lower()
    return any(marker in error_text for marker in (
        "remote end closed", "connection aborted", "remotedisconnected",
        "connection refused", "max retries exceeded", "newconnectionerror",
        "invalid session id", "chrome not reachable", "disconnected",
        "tab crashed", "session deleted", "cannot connect to chrome",
        "session not created",
    ))

### DB_Beamin_01_now.py (복원 대상 함수)
```python
def collect_now_for_driver(driver, account_id: str, store_list: list[dict]) -> None:
    """이미 로그인된 driver로 now 지표를 수집한다 (login/logout 없음)."""
    for store_info in store_list:
        try:
            if not is_on_main_dashboard(driver.current_url):
                driver.set_page_load_timeout(45)
                driver.get("https://self.baemin.com/")
            if not select_store_by_id(driver, store_info["store_id"]):
                raise RuntimeError(f"store selection failed: {store_info['store_id']}")
            if not wait_for_metrics_data(driver, timeout=90):
                raise RuntimeError(f"metrics load timeout: {store_info['store_id']}")
            stats = collect_single_store_stats(driver, store_info["store_id"], account_id)
            stats["brand"] = store_info["brand"]; stats["store"] = store_info["store"]
            _save_metrics_csv(stats, store_info["brand"], store_info["store"])
        except Exception as e:
            if _is_recoverable_driver_error(e):   # 렌더러/세션 에러는 상위로 전파
                raise
            logger.warning("매장 now 수집 실패 (건너뜀): %s / %s", store_info["store"], e)
```
