### 1) `content/02_baemin.js` — collect(opts) 배치모드 + 완료신호
`Sites['baemin'].collect()`(현재 `:10 async collect()`, 인자 없음)를 `collect(opts = {})`로 변경:
- 첫머리: `this._batchMode = opts.source === 'batch';`
- 비-도리당 매장 `confirm()` 블록(:24-33)을 `if (!this._batchMode) { ... }`로 감싸 배치모드에서 건너뜀.
- URL 라우팅 분기(:36-48)는 그대로 두되, 분기 진입 시 `category` 변수 결정:
  - `/orders/history`→`'orders'`, `/stat/marketing/woori-shop-click`→`'woori'`, `/stat/advertisement`→`'ad'`, `/history/change/shop`→`'change'`, 그 외→`'now'`.
- 라우팅 호출 전체를 `try { ... } catch(e){ success=false } finally { ... }`로 감싸고, **finally에서**:
  ```js
  if (this._batchMode) {
    try { chrome.runtime.sendMessage({ type:'BAEMIN_CATEGORY_COMPLETE',
  ```

```js
async function injectContent(tabId) {
  try { await chrome.tabs.sendMessage(tabId, {type:'PING'}); return true; }
  catch(_) {
    try { await chrome.scripting.executeScript({target:{tabId}, files:CONTENT_FILES});
      await sleep(500); return true; }

// 완료 신호 수신 (배민은 BAEMIN_CATEGORY_COMPLETE로 교체)
chrome.runtime.onMessage.addListener((msg) => {
  if (msg?.type === 'LIVE_LOG') { log(msg.msg); return; }
  if (!pending || pending.settled) return;
  if (msg?.type === 'MULTISTORE_COMPLETE') {
    pending.settled = true; pending.payload = msg.payload || {};
    clearTimeout(pending._to); clearInterval(pending._poll); pending.resolve();
  }
});
function waitMultistore(category, timeoutMs = COLLECT_TIMEOUT) {
  return new Promise(resolve => {
    const obj = { settled:false, payload:{}, resolve, _to:null, _poll:null }; pending = obj;
    obj._poll = setInterval(() => { if (stopRequested && !obj.settled) {
      obj.settled=true; obj.payload={category,stopped:true};
      clearInterval(obj._poll); clearTimeout(obj._to); resolve(); } }, 500);
    obj._to = setTimeout(() => { if (!obj.settled) {
      obj.settled=true; obj.payload={category,timeout:true};
      clearInterval(obj._poll); resolve(); } }, timeoutMs);
  });
}
```
