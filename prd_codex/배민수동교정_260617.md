```python
def reconcile_baemin_for_test_stores(
    stores: list[str],
    sale_date: str | None = None,
    lookback_days: int = 7,
) -> str:
    """
    TEST_STORES의 배달의민족 행을 baemin_macro 직수집 기준으로 교정.
    - 직수집 파일 없는 날짜 → 기존 posfeed 보존
    - 직수집 파일 있어도 해당 날짜 행 0건 → 스킵
    """
    if sale_date:
        dates = [sale_date]
    else:
        kst_now = pendulum.now("Asia/Seoul")
        dates = [
            (kst_now - timedelta(days=i)).strftime("%Y-%m-%d")
            for i in range(lookback_days)
        ]
    ym_list = sorted(set(d[:7] for d in dates))

    total_added = 0
    total_removed = 0

    store_map = _load_store_map()

    for store in stores:
        for ym in ym_list:
            # ym 내 파일 탐색 (brand=* glob)
            baemin_files = _find_baemin_files(store, ym)
            if not baemin_files:
                logger.info("배민 수집 파일 없음 (스킵): store=%s ym=%s", store, ym)
                continue

            df_raw = _read_baemin_files(baemin_files)
            if df_raw.empty:
                logger.info("배민 수집 데이터 없음 (스킵): store=%s ym=%s", store, ym)
                continue

            # brand는 경로에서 추출
            brand = _parse_brand_from_path(baemin_files[0])

            # 날짜별로 처리
            df_raw["sale_date"], df_raw["order_time"] = zip(
                *df_raw["주문시각"].map(_parse_baemin_datetime)
            )

            for date in dates:
                if date[:7] != ym:
                    continue
                df_day = df_raw[df_raw["sale_date"] == date].copy()
                if df_day.empty:
                    logger.info("배민 데이터 없음 → posfeed 보존: store=%s date=%s", store, date)
                    continue

                df_unified = _transform_to_unified(df_day, store, brand, store_map)
                df_unified = _apply_posfeed_blacklist(df_unified)

                removed, added = _upsert_daily(df_unified, date, store)
                total_removed += removed
                total_added += added
                logger.info(
                    "교정 완료: store=%s date=%s | 제거=%d 추가=%d",
                    store, date, removed, added,
                )

    return f"배민수동 교정 완료 | 제거={total_removed}행 추가={total_added}행"
```

#### 내부 함수들

```python
def _find_baemin_files(store: str, ym: str) -> list[str]:
    """brand=* glob으로 baemin_macro orders 파일 탐색 (.parquet 우선, .csv 폴백)."""
    base = ANALYTICS_DB / "baemin_macro" / "orders"
    for ext in ("parquet", "csv"):
        pattern = str(base / f"brand=*" / f"store={store}" / f"ym={ym}" / f"orders_{ym}.{ext}")
        files = sorted(glob(pattern))
        if files:
            return files
    return []


def _read_baemin_files(files: list[str]) -> pd.DataFrame:
    dfs = []
    for f in files:
        try:
            df = pd.read_parquet(f) if f.endswith(".parquet") else pd.read_csv(f, dtype=str)
            df["_src_path"] = f
            dfs.append(df)
        except Exception as e:
            logger.warning("배민 파일 읽기 실패 (%s): %s", f, e)
    return pd.concat(dfs, ignore_index=True) if dfs else pd.DataFrame()


def _parse_brand_from_path(file_path: str) -> str:
    """'...brand=도리당/store=...' 경로에서 brand 추출."""
    m = re.search(r"brand=([^/\\]+)", file_path)
    return m.group(1) if m else ""


def _parse_baemin_datetime(s: str) -> tuple[str, str]:
    """
    "2026. 06. 15. (월) 오후 10:48:04" → ("2026-06-15", "22:48:04")
    """
    if not isinstance(s, str):
        return "", ""
    m = re.match(
        r"(\d{4})\. (\d{2})\. (\d{2})\. \(.+\) (오전|오후) (\d{1,2}):(\d{2}):(\d{2})", s
    )
    if not m:
        return "", ""
    year, month, day = m.group(1), m.group(2), m.group(3)
    ampm, hour, minute, second = m.group(4), int(m.group(5)), m.group(6), m.group(7)
    if ampm == "오후" and hour != 12:
        hour += 12
    elif ampm == "오전" and hour == 12:
        hour = 0
    return f"{year}-{month}-{day}", f"{hour:02d}:{minute}:{second}"


def _transform_to_unified(
    df: pd.DataFrame,
    store: str,
    brand: str,
    store_map: dict,
) -> pd.DataFrame:
    """baemin_macro rows → UNIFIED_COLUMNS DataFrame."""
    out = pd.DataFrame()

    out["sale_date"]        = df["sale_date"]
    out["ym"]               = df["sale_date"].str[:7]
    out["source"]           = "배민수동"
    out["brand"]            = brand
    out["store"]            = store
    out["platform"]         = "배달의민족"
    out["order_id"]         = df["주문번호"].astype(str).str.strip()
    out["order_time"]       = df["order_time"]
    out["sale_type"]        = df["주문상태"].apply(
        lambda x: "취소" if "취소" in str(x) else "정상"
    )
    out["order_type"]       = df["수령방법"].apply(
        lambda x: "배달_포장" if str(x).strip() == "포장" else "배달"
    )
    # 주문내역: " 외 N건" suffix 제거
    out["menu_name"]        = (
        df["주문내역"].fillna("").astype(str)
        .str.replace(r"\s*외\s*\d+건$", "", regex=True)
        .str.strip()
    )
    out["item_name"]        = df["주문옵션상세"].fillna("").astype(str).str.strip()
    out["qty"]              = pd.to_numeric(df["주문수량"], errors="coerce").fillna(1).astype(int)
    out["unit_price"]       = pd.to_numeric(df["주문옵션금액"], errors="coerce").fillna(0).astype(int)
    out["discount_amount"]  = 0

    # item_seq: baemin 원본 행 순서 그대로 (자연 순서 = 메인 아이템 먼저)
    out["item_seq"] = (
        out.groupby("order_id").cumcount().add(1).astype(int).astype(str)
    )

    # total_price: 상품금액 비율 배분
    out["item_total"]       = out["unit_price"] * out["qty"]
    order_item_sum          = out.groupby("order_id")["item_total"].transform("sum")
    baemin_상품금액         = (
        pd.to_numeric(df["상품금액"], errors="coerce").fillna(0).astype(int)
    )
    out["_order_상품금액"]  = baemin_상품금액.values
    order_상품금액          = out.groupby("order_id")["_order_상품금액"].transform("first")
    out["total_price"]      = (
        (out["item_total"] / order_item_sum.replace(0, 1)) * order_상품금액
    ).round().astype(int)

    # order_cnt: 주문 당 unit_price 최대 행 → 정상:1 / 취소:-1, 나머지 0
    out["order_cnt"] = 0
    max_idx = out.groupby("order_id")["unit_price"].idxmax()
    out.loc[max_idx, "order_cnt"] = out.loc[max_idx, "sale_type"].map(
        {"정상": 1, "취소": -1}
    )

    # item_id: MD5(item_name)
    out["item_id"] = out["item_name"].map(
        lambda s: hashlib.md5(s.encode()).hexdigest() if s else ""
    )

    # store 메타
    out["담당자"]   = _lookup_store_meta(store_map, store, "담당자")
    out["region"]   = _lookup_store_meta(store_map, store, "region")
    out["실오픈일"] = _lookup_store_meta(store_map, store, "실오픈일")

    out["collected_at"] = pendulum.now("Asia/Seoul").isoformat()

    # _pk
    out["_pk"] = _make_unified_pk(out)

    # 임시 컬럼 제거
    out = out.drop(columns=["item_total", "_order_상품금액"], errors="ignore")

    return out.reindex(columns=UNIFIED_COLUMNS, fill_value="")


def _upsert_daily(df_new: pd.DataFrame, date: str, store: str) -> tuple[int, int]:
    """
    unified_sales 일별 parquet에서 해당 store의 배달의민족 행을 교체.

    Returns:
        (removed_count, added_count)
    """
    from modules.transform.pipelines.db.DB_UnifiedSales_common import UNIFIED_ROOT
    UNIFIED_ROOT.mkdir(parents=True, exist_ok=True)
    daily_path = _unified_daily_path(date)

    if daily_path.exists():
        df_existing = pd.read_parquet(daily_path).reindex(columns=UNIFIED_COLUMNS, fill_value="")
        # 멱등성 보장: posfeed 배민 행 + 이전 배민수동 행 모두 제거
        remove_mask = (
            (df_existing["store"] == store) &
            (df_existing["platform"] == "배달의민족") &
            (df_existing["source"].isin(["posfeed", "배민수동"]))
        )
        removed_count = int(remove_mask.sum())
        df_existing = df_existing[~remove_mask]
    else:
        df_existing = pd.DataFrame(columns=UNIFIED_COLUMNS)
        removed_count = 0

    df_out = pd.concat([df_existing, df_new], ignore_index=True)

    # _pk 중복 방어
    if "_pk" in df_out.columns:
        before = len(df_out)
        df_out = df_out.drop_duplicates(subset=["_pk"], keep="last").reset_index(drop=True)
        if len(df_out) < before:
            logger.warning("_pk 중복 %d행 제거", before - len(df_out))

    # int 컬럼 보정
    for col in ("unit_price", "total_price", "discount_amount"):
        if col in df_out.columns:
            df_out[col] = pd.to_numeric(df_out[col], errors="coerce").fillna(0).astype(int)

    df_out.to_parquet(daily_path, index=False, engine="pyarrow")
    return removed_count, len(df_new)
```

## Constraints

- `source = "배민수동"` 고정 — 다른 값 사용 금지
- `platform = "배달의민족"` 고정
- parquet 경로는 반드시 `_unified_daily_path()` 사용 (2자리 연도 `%y%m%d` 포맷)
- brand 하드코딩 금지 — glob 경로 `brand=*` 파티션명에서 추출
- baemin 파일 없거나 해당 날짜 0건이면 기존 posfeed 데이터 보존 (제거 금지)
- `_apply_posfeed_blacklist()` 반드시 적용 — whitelist is_valid='N' 항목 제거
- item_seq는 baemin 원본 행 순서 그대로 (sort 금지)
- total_price는 `상품금액` 컬럼 비율 배분 (배달비 제외)
- task 실패 시 DAG 전체 실패 (on_failure_callback 상속)
- 멱등성 보장: `source.isin(['posfeed', '배민수동'])` 행 모두 제거 후 재삽입
- `TEST_STORES = []` 이면 task는 즉시 "스킵" 반환 (에러 아님)
