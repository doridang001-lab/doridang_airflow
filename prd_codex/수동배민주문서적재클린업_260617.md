```python
"""수동 배치 배민 주문 CSV 적재 파이프라인."""
from __future__ import annotations

import json
import logging
import re
import shutil
from pathlib import Path

import pandas as pd
import pendulum

from modules.transform.pipelines.db.DB_Beamin_Macro_validate import (
    _manual_baemin_store_meta,
    _manual_baemin_filename_fallback,
)
from modules.transform.pipelines.db.DB_Beamin_04_orders import _COLUMNS
from modules.transform.pipelines.db.beamin_store_io import read_table, write_table
from modules.transform.utility.paths import COLLECT_DB, BAEMIN_ORDERS_DB

logger = logging.getLogger(__name__)

COLLECT_SRC = COLLECT_DB / "영업관리부_수집"
ARCHIVE_DIR = COLLECT_SRC / "_archived"
_REQUIRED = {"주문상태", "주문번호", "주문시각", "결제금액"}


def _ym_from_order_time(value: str) -> str:
    """'2026. 06. 17. 오전...' → '2026-06'"""
    m = re.match(r"(\d{4})\.\s*(\d{2})\.", str(value).strip())
    if not m:
        return ""
    return f"{m.group(1)}-{m.group(2)}"


def _archive_path(path: Path) -> Path:
    ARCHIVE_DIR.mkdir(parents=True, exist_ok=True)
    target = ARCHIVE_DIR / path.name
    if not target.exists():
        return target
    stem, suffix = path.stem, path.suffix
    idx = 1
    while True:
        candidate = ARCHIVE_DIR / f"{stem}.{idx}{suffix}"
        if not candidate.exists():
            return candidate
        idx += 1


def _load_one_file(csv_path: Path) -> tuple[int, bool]:
    """단일 CSV 로드 → 적재 행 수 반환. 실패 시 (0, False)."""
    for enc in ("utf-8-sig", "cp949"):
        try:
            df = pd.read_csv(csv_path, dtype=str, encoding=enc)
            break
        except (UnicodeDecodeError, Exception):
            df = None
    if df is None or df.empty:
        logger.warning("읽기 실패 또는 빈 파일: %s", csv_path.name)
        return 0, False

    if not _REQUIRED.issubset(df.columns):
        missing = _REQUIRED - set(df.columns)
        logger.warning("필수 컬럼 없음 %s: %s", csv_path.name, missing)
        return 0, False

    fallback = _manual_baemin_filename_fallback(csv_path)
    raw_store = ""
    if "store_name" in df.columns and not df["store_name"].dropna().empty:
        raw_store = str(df["store_name"].dropna().astype(str).iloc[0])
    store_key, brand = _manual_baemin_store_meta(raw_store, fallback)
    if not store_key or not brand:
        logger.warning("매장 파싱 실패: %s / raw=%s", csv_path.name, raw_store)
        return 0, False

    target_df = df[
        (df["주문상태"].astype(str) == "배달완료")
        & df["결제금액"].astype(str).str.strip().ne("")
    ].copy()
    if target_df.empty:
        logger.info("배달완료 행 없음: %s", csv_path.name)
        return 0, True  # 파일은 정상, 데이터만 없음 → 아카이브 대상

    now_str = pendulum.now("Asia/Seoul").isoformat()
    target_df["collected_at"] = now_str
    target_df["store_name"] = store_key

    # ym별 그룹핑 → 파티션 upsert
    target_df["_ym"] = target_df["주문시각"].apply(_ym_from_order_time)
    total_rows = 0
    for ym, group in target_df.groupby("_ym"):
        if not ym:
            logger.warning("ym 추출 실패 행 %d개 건너뜀: %s", len(group), csv_path.name)
            continue
        new_df = pd.DataFrame("", index=group.index, columns=_COLUMNS, dtype=str)
        for col in _COLUMNS:
            if col in group.columns:
                new_df[col] = group[col].astype(str).values
        new_df = new_df.fillna("").astype(str)

        stem = BAEMIN_ORDERS_DB / f"brand={brand}" / f"store={store_key}" / f"ym={ym}" / f"orders_{ym}"
        new_order_ids = set(new_df["주문번호"].unique())
        existing = read_table(stem)
        if existing is not None:
            existing = existing[~existing["주문번호"].isin(new_order_ids)]
            combined = pd.concat([existing, new_df], ignore_index=True)
        else:
            combined = new_df
        out_path = write_table(combined, stem)
        total_rows += len(new_df)
        logger.info("저장: %s/%s ym=%s → %d행 (%s)", brand, store_key, ym, len(new_df), out_path)

    return total_rows, True


def load_manual_baemin_orders(**context) -> str:
    """영업관리부_수집/baemin_orders_*.csv → baemin_macro/orders 파티션 적재."""
    files = sorted(COLLECT_SRC.glob("baemin_orders_*.csv"))
    if not files:
        logger.info("처리할 파일 없음: %s", COLLECT_SRC)
        return json.dumps({"loaded_files": [], "skipped_files": []}, ensure_ascii=False)

    loaded, skipped = [], []
    for csv_path in files:
        rows, ok = _load_one_file(csv_path)
        if ok:
            loaded.append(str(csv_path))
            logger.info("적재 완료: %s (%d행)", csv_path.name, rows)
        else:
            skipped.append(str(csv_path))
            logger.warning("적재 실패(skip): %s", csv_path.name)

    return json.dumps(
        {"loaded_files": loaded, "skipped_files": skipped},
        ensure_ascii=False,
    )


def cleanup_manual_baemin_orders(**context) -> str:
    """ingest 태스크에서 성공한 파일을 _archived/ 로 이동."""
    ti = context["task_instance"]
    raw = ti.xcom_pull(task_ids="ingest_manual_baemin_orders", key="return_value")
    if not raw:
        logger.info("XCom 없음, cleanup 건너뜀")
        return "이동 완료: 0개"

    result = json.loads(raw)
    loaded_files = result.get("loaded_files", [])
    moved = 0
    for path_str in loaded_files:
        path = Path(path_str)
        if not path.exists():
            logger.info("이미 없음: %s", path.name)
            continue
        target = _archive_path(path)
        try:
            shutil.move(str(path), str(target))
            logger.info("아카이브: %s → %s", path.name, target)
            moved += 1
        except Exception as exc:
            logger.warning("이동 실패: %s / %s", path.name, exc)

    return f"이동 완료: {moved}개"
```

```python
def _manual_baemin_store_meta(raw_store_name: str, fallback_name: str = "") -> tuple[str, str]:
    """Returns (store_key, brand) parsed from a manual Baemin CSV source name."""
    text = str(raw_store_name or fallback_name or "").strip()
    if not text:
        return "", ""
    text = re.sub(r"\[.*?\]\s*", "", text).strip()
    brand = "나홀로" if "나홀로" in text else ("도리당" if "도리당" in text else "")
    matches = re.findall(r"[가-힣A-Za-z0-9]+(?:점|지점|분점|직영점)", text)
    branch = matches[-1] if matches else (text.split()[-1] if text.split() else text)
    normalized = f"{brand} {branch}".strip() if brand else branch
    normalized_series = normalize_store_names(pd.Series([normalized]))
    branch_series = strip_brand(normalized_series)
    return str(branch_series.iloc[0]).strip(), brand

def _manual_baemin_filename_fallback(csv_path: Path) -> str:
    stem = csv_path.stem
    stem = re.sub(r"^baemin_orders_", "", stem)
    stem = re.sub(r"_unknown_\d{8}$", "", stem)
    return stem.replace("_", " ").strip()
```

### beamin_store_io.py (read_table / write_table)
```python
def read_table(stem_path: Path, columns=None) -> pd.DataFrame | None:
    pq = stem_path.with_suffix(".parquet")
    csv = stem_path.with_suffix(".csv")
    if pq.exists():
        return _as_str_df(pd.read_parquet(pq, columns=columns))
    if csv.exists():
        return _as_str_df(pd.read_csv(csv, dtype=str, encoding="utf-8-sig", usecols=columns))
    return None

def write_table(df: pd.DataFrame, stem_path: Path) -> Path:
    stem_path.parent.mkdir(parents=True, exist_ok=True)
    pq = stem_path.with_suffix(".parquet")
    _as_str_df(df).to_parquet(pq, engine="pyarrow", compression="zstd", index=False)
    csv = stem_path.with_suffix(".csv")
    if csv.exists():
        csv.unlink()
    return pq
```

2. **DAG import 검증** `python -c "from dags.db.DB_CollectionCompare_Dags import dag"` → ImportError 없음

4. **실제 파일 적재** `baemin_orders_*.csv` 1개 올려두고 DAG manually trigger → `ingest` 로그에 "저장:" 확인, `BAEMIN_ORDERS_DB/brand=.../store=.../ym=.../orders_{ym}.parquet` 생성 확인
5. **cleanup 검증** `cleanup` 태스크 완료 후 `영업관리부_수집/_archived/` 에 파일 이동 확인, 원본 삭제 확인
6. **build_compare 연속 실행** `ingest >> cleanup >> build_compare` 전체 체인 GREEN 확인

## Verification Loop

```
LOOP until all PASS:
  1. Test Cases 순서대로 실행
  2. FAIL 항목 → 원인 분석
  3. 코드 수정
  4. 전체 Test Cases 재실행
종료 조건: 전체 PASS + Constraints 위반 없음
```

## Constraints

- `_manual_baemin_store_meta`, `_manual_baemin_filename_fallback` 는 `DB_Beamin_Macro_validate.py`에서 직접 import (재구현 금지)
- `_COLUMNS` 는 `DB_Beamin_04_orders.py`에서 직접 import (하드코딩 금지)
- `read_table`, `write_table` 은 `beamin_store_io.py` 사용 (parquet zstd 저장 보장)
- cleanup 은 성공한 파일만 아카이브(`loaded_files`). `skipped_files` 는 건드리지 않음
- 기존 `build_compare` task_id 변경 금지
- 태스크 의존성: `ingest_baemin >> cleanup_baemin >> build_compare` 순서 고정
