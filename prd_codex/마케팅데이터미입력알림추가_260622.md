```python
def check_marketing_input(**context) -> None:
    """hall_marketing_target.csv 최근 7일 미입력 날짜 감지 → HTML 이메일 알림."""
    from datetime import date, timedelta

    import pandas as pd

    from modules.transform.utility.mailer import send_email

    today = date.today()
    check_dates = sorted([today - timedelta(days=i) for i in range(1, 8)])

    if not MKT_CSV.exists():
        logger.warning("hall_marketing_target.csv 없음: %s", MKT_CSV)
        missing = [(d, "CSV 파일 없음") for d in check_dates]
        html = _build_missing_html(missing, str(MKT_CSV))
        subject = f"[마케팅 데이터 미입력] CSV 파일 없음 — 확인 요청"
        send_email(subject=subject, html_content=html, to_emails=_ALERT_TO, cc_emails=_ALERT_CC)
        return

    df = pd.read_csv(MKT_CSV, dtype=str)
    date_col = df.get("기준일자") if "기준일자" in df.columns else None
    fallback = df.get("입력날짜") if "입력날짜" in df.columns else None
    df["_기준일"] = pd.to_datetime(date_col, errors="coerce")
    if fallback is not None:
        df["_기준일"] = df["_기준일"].fillna(pd.to_datetime(fallback, errors="coerce"))

    for col in _MKT_COLS:
        if col in df.columns:
            df[col] = pd.to_numeric(
                df[col].astype(str).str.replace(",", "", regex=False), errors="coerce"
            ).fillna(0)

    entered = set(df["_기준일"].dt.date.dropna())
    missing = []
    for d in check_dates:
        if d not in entered:
            missing.append((d, "행 없음"))
            continue
        row = df[df["_기준일"].dt.date == d]
        total = sum(row[c].sum() for c in _MKT_COLS if c in row.columns)
        if total == 0:
            missing.append((d, "전체 항목 0"))

    if not missing:
        logger.info("마케팅 입력 확인 완료 — 미입력 날짜 없음 (최근 7일)")
        return

    missing_dates_str = ", ".join(d.strftime("%m/%d") for d, _ in missing)
    subject = f"[마케팅 데이터 미입력] {missing_dates_str} 입력 요청"
    html = _build_missing_html(missing, str(MKT_CSV))
    send_email(subject=subject, html_content=html, to_emails=_ALERT_TO, cc_emails=_ALERT_CC)
    logger.info("마케팅 미입력 알림 발송: %s", missing_dates_str)
```

```python
def send_email(
    subject,
    html_content,
    to_emails,
    conn_id="doridang_conn_smtp_gmail",
    attachments=None,
    inline_images=None,
    **context,
):
    del context
    import smtplib, time
    from email.mime.application import MIMEApplication
    from email.mime.image import MIMEImage
    from email.mime.multipart import MIMEMultipart
    from email.mime.text import MIMEText
    from airflow.hooks.base import BaseHook

    connection = BaseHook.get_connection(conn_id)
    smtp_host = connection.host
    smtp_port = int(connection.port)
    smtp_user = connection.login
    smtp_password = connection.password
    extra = connection.extra_dejson
    from_email = extra.get("from_email") or smtp_user
    timeout = int(extra.get("timeout") or 30)
    retry_limit = max(1, int(extra.get("retry_limit") or 1))

    to_list = [to_emails] if isinstance(to_emails, str) else list(to_emails)
    attachments = list(attachments or [])
    inline_images = list(inline_images or [])

    msg = MIMEMultipart("related")
    msg["Subject"] = subject
    msg["From"] = from_email
    msg["To"] = ", ".join(to_list)
    # ↑ 여기 아래에 cc_list/msg["Cc"] 삽입

    alternative_part = MIMEMultipart("alternative")
    alternative_part.attach(MIMEText(html_content, "html", "utf-8"))
    msg.attach(alternative_part)
    # ... inline_images, attachments 루프 ...

    for attempt in range(1, retry_limit + 1):
        try:
            if smtp_port == 465:
                with smtplib.SMTP_SSL(smtp_host, smtp_port, timeout=timeout) as server:
                    server.login(smtp_user, smtp_password)
                    server.send_message(msg)   # ← To/Cc 헤더 자동 반영
            else:
                with smtplib.SMTP(smtp_host, smtp_port, timeout=timeout) as server:
                    server.ehlo(); server.starttls(); server.ehlo()
                    server.login(smtp_user, smtp_password)
                    server.send_message(msg)
            return f"메일 발송 완료: {len(to_list)}명"
        except Exception as exc:
```

## Constraints

- `send_email()`의 기존 호출부(cc_emails 미사용)는 모두 그대로 동작해야 함 (`cc_emails=None` 기본값 필수)
- `server.send_message(msg)` 유지 (sendmail로 교체 금지) — CC 헤더만 추가하면 자동 반영됨
- `check_dates` 범위는 `today - 1일 ~ today - 7일` (today 당일 제외)
- 마케팅 CSV가 없을 때도 예외 없이 알림 메일 발송 후 정상 종료
- `logger` 인스턴스는 이미 파일 상단에 선언되어 있음 — 재선언 금지
